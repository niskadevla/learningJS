Task 4.7.11: Решето Эратосфена 

  Целое число, большее 1, называется простым, если оно не делится нацело ни на какое другое, кроме себя и 1.

  Древний алгоритм «Решето Эратосфена» для поиска всех простых чисел до n выглядит так:

      Создать список последовательных чисел от 2 до n: 2, 3, 4, ..., n.
      Пусть p=2, это первое простое число.
      Зачеркнуть все последующие числа в списке с разницей в p, т.е. 2*p, 3*p, 4*p и т.д. В случае p=2 это будут 4,6,8....
      Поменять значение p на первое не зачеркнутое число после p.
      Повторить шаги 3-4 пока p2 < n.
      Все оставшиеся не зачеркнутыми числа – простые.

  Посмотрите также анимацию алгоритма.

  Реализуйте «Решето Эратосфена» в JavaScript, используя массив.

  Найдите все простые числа до 100 и выведите их сумму.

Решение


  My solution
  'use strict';

  var arrayNumber = [];
  var arr = [];
  var arrayNumberNew = [];

  var i, j, p, p2, n, count;

  // Создали массив чисел от 2 до n
  n = 101;
  p = 2;
  count = 0;
  for ( i = 2; i <= n; i++ ) {
    arrayNumber[i - 2] = i;
  }


  // Главный цикл
  while (p*p < n) {

    i = p;
    p2 = i * p;

    //Создаем массив не нужных чисел
    while (p2 <= n) {
      arr.push( arrayNumber[ arrayNumber.indexOf(p2) ] );
      i++;
      p2 = i * p;
    }

    // Создаем массив нужных чисел
    for ( j = 0; j < arrayNumber.length; j++ ) {
      if (  arr.indexOf(arrayNumber[j]) != -1  ) continue;
      else arrayNumberNew.push( arrayNumber[j] );
    }

    count++; // Счетчик
    n = arrayNumberNew[arrayNumberNew.length - 1]; // Последний элемент массива
    p = arrayNumberNew[count]; // Следующий элемент в массиве
    arr.length = 0;
    arrayNumber.length = 0; // Обнуляем входной массив

    // Переприсваеваем входной массив (становится новым массивом)
    for (j = 0; j < arrayNumberNew.length; j++) {
      arrayNumber.push( arrayNumberNew[j] );
    }

    arrayNumberNew.length = 0;

  }

  //Сумма чисел Эратосфена
  var sum = 0;
  for (i = 0; i < arrayNumber.length; i++) {
    sum += arrayNumber[i];
  }

  function camelize(str) {
    var arr = str.split('');
    for (var i = 0; i < arr.length; i++) {
      if ( arr[i] == "-") {
        arr[i + 1] = arr[i + 1].toUpperCase();
        arr.splice(i,1);
      }
    }

    return arr.join('');

  'use strict';

  var arr = [1,2,3,4,5];

  function getSums(arr) {

      arrSum.push(previousValue);
      return previousValue + currentItem;
    });

    arrSum.push(sum);

    return arrSum;
  }
